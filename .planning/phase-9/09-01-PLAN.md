    ```tsx
    // Load Ammo.js (typically from CDN)
    await Ammo()

    // Create physics world
    const collisionConfig = new Ammo.btDefaultCollisionConfiguration()
    const dispatcher = new Ammo.btCollisionDispatcher(collisionConfig)
    const broadphase = new Ammo.btDbvtBroadphase()
    const solver = new Ammo.btSequentialImpulseConstraintSolver()
    const world = new Ammo.btDiscreteDynamicsWorld(dispatcher, broadphase, solver, collisionConfig)
    world.setGravity(new Ammo.btVector3(0, -9.81, 0))

    // Create rigid body
    const shape = new Ammo.btBoxShape(new Ammo.btVector3(1, 1, 1))
    const transform = new Ammo.btTransform()
    transform.setIdentity()
    transform.setOrigin(new Ammo.btVector3(0, 10, 0))

    const motionState = new Ammo.btDefaultMotionState(transform)
    const rbInfo = new Ammo.btRigidBodyConstructionInfo(mass, motionState, shape, localInertia)
    const body = new Ammo.btRigidBody(rbInfo)
    world.addRigidBody(body)
    ```

    ```tsx
    // Create soft body for cloth
    const softBodyHelpers = new Ammo.btSoftBodyHelpers()
    const clothSoftBody = softBodyHelpers.CreatePatch(
        world.getWorldInfo(),
        corners,
        segmentsX,
        segmentsZ,
        fixedPoints
    )
    world.addSoftBody(clothSoftBody)

    // Sync cloth vertices with Three.js mesh
    useFrame(() => {
        const nodes = clothSoftBody.get_m_nodes()
        const positions = geometry.attributes.position.array
        for (let i = 0; i < nodes.size(); i++) {
            const node = nodes.at(i)
            const p = node.get_m_x()
            positions[i * 3] = p.x()
            positions[i * 3 + 1] = p.y()
            positions[i * 3 + 2] = p.z()
        }
        geometry.attributes.position.needsUpdate = true
    })
    ```

    Rapier integration pattern:
    ```tsx
    import RAPIER from '@dimforge/rapier3d'

    // Initialize Rapier
    const gravity = { x: 0.0, y: -9.81, z: 0.0 }
    const world = new RAPIER.World(gravity)

    // Create rigid body
    const bodyDesc = RAPIER.RigidBodyDesc.dynamic()
        .setTranslation(0.0, 10.0, 0.0)
    const body = world.createRigidBody(bodyDesc)

    // Create collider
    const colliderDesc = RAPIER.ColliderDesc.cuboid(1.0, 1.0, 1.0)
    const collider = world.createCollider(colliderDesc, body)

    // Step simulation
    useFrame(() => {
        world.step()

        // Sync with Three.js mesh
        const position = body.translation()
        const rotation = body.rotation()
        mesh.position.set(position.x, position.y, position.z)
        mesh.quaternion.set(rotation.x, rotation.y, rotation.z, rotation.w)
    })
    ```
