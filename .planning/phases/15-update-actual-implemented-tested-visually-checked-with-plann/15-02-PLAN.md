Canonical init3D module scaffold:
  /** @jsxImportSource woby */
  import * as THREE from 'three'
  import Stats from 'three/examples/jsm/libs/stats.module.js'

  let _cleanupFn: (() => void) | null = null

  const init3D = (container: HTMLElement) => {
    if (_cleanupFn) { _cleanupFn(); _cleanupFn = null }
    let SCREEN_WIDTH = container.clientWidth || window.innerWidth
    let SCREEN_HEIGHT = container.clientHeight || window.innerHeight
    // ... scene/camera/renderer setup ...
    renderer = new THREE.WebGLRenderer({ antialias: true })
    renderer.setPixelRatio(window.devicePixelRatio)
    renderer.setSize(SCREEN_WIDTH, SCREEN_HEIGHT)
    container.appendChild(renderer.domElement)
    // ... animate loop ...
    _cleanupFn = () => { cancelAnimationFrame(animId); renderer.dispose(); /* + remove DOM + listeners + gui.destroy() + controls.dispose() */ }
  }

  export const <Name> = () => (
    <div style="width:100%;height:100%;position:relative">
      <div ref={(el: HTMLElement | null) => { if (el) init3D(el) }} style="width:100%;height:100%" />
    </div>
  )
  export default <Name>

DemoEntry registry shape (from registry.ts):
  { id: string; name: string; category: string; component: () => Promise<{ default: () => JSX.Element }>; thumbnail?: string }

GUI cleanup contract: import `GUI` from `three/examples/jsm/libs/lil-gui.module.min.js`, construct with `new GUI({ container })`, call `gui.destroy()` in cleanup.
OrbitControls cleanup contract: `new OrbitControls(camera, renderer.domElement)`, call `controls.dispose()` in cleanup.
Type-safe material access (no `as any`): `(mesh.material as THREE.Material & { opacity: number }).opacity = 0.5`
Asset path convention: relative paths resolve to `http://localhost:5173/`; assets in `demo/public/` (e.g. `'models/gltf/Foo.glb'`).
