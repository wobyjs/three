    Implementation pattern (reference existing demo/src/WebGLInteractiveCubes.tsx):
    ```tsx
    /** @jsxImportSource @woby/three */

    import { $, $$, useEffect } from "woby"
    import { useFrame, useThree, useCamera } from '@woby/three'
    import * as THREE from 'three'

    export const WebGLInteractiveCubes = () => {
        const raycaster = new THREE.Raycaster()
        const pointer = new THREE.Vector2()
        const INTERSECTED = $<THREE.Mesh | null>(null)

        const onPointerMove = (event: PointerEvent) => {
            pointer.x = (event.clientX / window.innerWidth) * 2 - 1
            pointer.y = -(event.clientY / window.innerHeight) * 2 + 1
        }

        useFrame(() => {
            // CRITICAL: Use $$ for context arrays, not $
            const camera = $$(useThree().cameras)[0] as THREE.PerspectiveCamera
            const scene = $$(useThree().scene)
            raycaster.setFromCamera(pointer, camera)
            const intersects = raycaster.intersectObjects(scene!.children)

            // Update emissive color based on intersection
            const prev = $$(INTERSECTED)
            if (prev && prev.material) {
                (prev.material as THREE.MeshLambertMaterial).emissive.setHex(0x000000)
            }
            if (intersects.length > 0) {
                const hit = intersects[0].object as THREE.Mesh
                INTERSECTED(hit)
                if (hit.material) {
                    (hit.material as THREE.MeshLambertMaterial).emissive.setHex(0xff0000)
                }
            }
        })

        return (
            <canvas3D onpointermove={onPointerMove}>
                {/* 2000 cubes with raycasting */}
            </canvas3D>
        )
    }
    ```

    Critical considerations:
    - 2000 randomly positioned cubes

    - Camera orbit animation
    - Emissive color highlighting on hover
    - CRITICAL: Use `$$` for context arrays, never `$` (per memory feedback)
