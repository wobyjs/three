import * as https from 'https'

export function probeUrl(url: string): Promise<boolean> {
  return new Promise((resolve) => {
    const req = https.request(url, { method: 'HEAD', timeout: 8000 }, (res) => {
      resolve(res.statusCode !== undefined && res.statusCode < 400)
    })
    req.on('error', () => resolve(false))
    req.on('timeout', () => { req.destroy(); resolve(false) })
    req.end()
  })
}

**Probe-and-classify logic:** Since Playwright workers cannot share in-memory state, use a two-phase approach:
- Phase A (single describe block, `test.describe.configure({ mode: 'serial' })`): Run HTTP HEAD probes for all demos and write an initial manifest to disk.
- Phase B (parallel describe): Each demo in the manifest that has `reference_status: 'pending'` gets a `test()` entry that captures its screenshot and updates its manifest entry.

Implement as two describe blocks in the same file:

```typescript
// Phase A: Probe all demos and write initial manifest
test.describe('Reference probe phase', () => {
  test.describe.configure({ mode: 'serial' })

  test('Probe all demos and build initial manifest', async () => {
    const manifest: ReferenceEntry[] = []

    // Process in batches of 20 concurrent probes
    async function runWithConcurrency<T>(items: T[], n: number, fn: (item: T) => Promise<void>) {
      const q = [...items]
      const worker = async () => { while (q.length > 0) await fn(q.shift()!) }
      await Promise.all(Array.from({ length: Math.min(n, items.length) }, worker))
    }

    await runWithConcurrency(ALL_DEMOS, 20, async (demo) => {
      let entry: ReferenceEntry
      if (CUSTOM_DEMO_IDS.has(demo.id)) {
        entry = { ...demo, has_reference: false, reference_url: null, reference_status: 'no-reference', screenshot_path: null }
      } else {
        const url = `${THREEJS_BASE}/${demo.id}.html`
        const reachable = await probeUrl(url)
        if (reachable) {
          entry = { ...demo, has_reference: true, reference_url: url, reference_status: 'pending', screenshot_path: null }
        } else {
          entry = { ...demo, has_reference: false, reference_url: url, reference_status: 'http-404', screenshot_path: null }
        }
      }
      manifest.push(entry)
    })

    // Write initial manifest
    fs.writeFileSync(
      path.join(ROOT, REFERENCE_MANIFEST_PATH),
      JSON.stringify(manifest, null, 2)
    )
    console.log(`Probed ${manifest.length} demos. Reachable: ${manifest.filter(e => e.reference_status === 'pending').length}`)
    expect(manifest.length).toBeGreaterThan(0)
  })
})

// Phase B: Capture reference screenshots for all reachable demos
test.describe('Reference capture phase', () => {
  // Read manifest that Phase A wrote
  let pendingDemos: ReferenceEntry[] = []

  test.beforeAll(async () => {
    const manifestPath = path.join(ROOT, REFERENCE_MANIFEST_PATH)
    if (fs.existsSync(manifestPath)) {
      const allEntries: ReferenceEntry[] = JSON.parse(fs.readFileSync(manifestPath, 'utf-8'))
      pendingDemos = allEntries.filter(e => e.reference_status === 'pending')
    }
    console.log(`Capturing ${pendingDemos.length} reference screenshots...`)
  })

  // Generate one test per pending demo — Playwright workers will parallelize these
  // This runs AFTER Phase A's serial probe is complete
  for (const demo of ALL_DEMOS) {
    test(`Capture reference: ${demo.id}`, async ({ page }) => {
      // Check if this demo needs capture (might have been probed as no-reference)
      const manifestPath = path.join(ROOT, REFERENCE_MANIFEST_PATH)
      const allEntries: ReferenceEntry[] = JSON.parse(fs.readFileSync(manifestPath, 'utf-8'))
      const entry = allEntries.find(e => e.id === demo.id)

      if (!entry || entry.reference_status !== 'pending') {
        test.skip()
        return
      }

      const outPath = path.join(ROOT, REFERENCE_SCREENSHOT_DIR, `${demo.id}.png`)
      const url = `${THREEJS_BASE}/${demo.id}.html`

      // Navigate with networkidle, fall back to domcontentloaded on timeout
      try {
        await page.goto(url, { waitUntil: 'networkidle', timeout: 25000 })
      } catch {
        try {
          await page.goto(url, { waitUntil: 'domcontentloaded', timeout: 15000 })
        } catch {
          // Mark as failed and update manifest
          updateManifestEntry(manifestPath, demo.id, { reference_status: 'reference-load-failed', has_reference: false })
          return
        }
      }

      // Wait for WebGL to initialize and render at least one frame (5s per RESEARCH.md)
      await page.waitForTimeout(REFERENCE_WAIT_MS)

      // Check canvas presence
      const canvasCount = await page.locator('canvas').count()
      if (canvasCount === 0) {
        updateManifestEntry(manifestPath, demo.id, { reference_status: 'reference-load-failed', has_reference: false })
        console.log(`  [no-canvas] ${demo.id}`)
        return
      }

      // Pixel check: verify non-blank WebGL render
      const hasContent = await page.evaluate((): boolean => {
        const canvas = document.querySelector('canvas') as HTMLCanvasElement | null
        if (!canvas) return false
        const gl = (canvas.getContext('webgl2') || canvas.getContext('webgl')) as WebGLRenderingContext | null
        if (!gl) return true  // canvas exists but no GL context — assume content present
        try {
          const pixels = new Uint8Array(4 * 16 * 16)
          gl.readPixels(0, 0, 16, 16, gl.RGBA, gl.UNSIGNED_BYTE, pixels)
          for (let i = 3; i < pixels.length; i += 4) {
            if (pixels[i] > 0) return true
          }
        } catch { return true }
        return false
      })

      if (!hasContent) {
        updateManifestEntry(manifestPath, demo.id, { reference_status: 'reference-load-failed', has_reference: false })
        console.log(`  [blank-canvas] ${demo.id}`)
        return
      }

      // Screenshot the canvas element only — exclude UI chrome for cleaner comparison
      const canvas = page.locator('canvas').first()
      await canvas.screenshot({ path: outPath })
      const relPath = path.relative(ROOT, outPath).replace(/\\/g, '/')
      updateManifestEntry(manifestPath, demo.id, {
        reference_status: 'ok',
        has_reference: true,
        screenshot_path: relPath
      })
      console.log(`  [ok] ${demo.id}`)
    })
  }

  test.afterAll(async () => {
    // Print summary
    const manifestPath = path.join(ROOT, REFERENCE_MANIFEST_PATH)
    if (fs.existsSync(manifestPath)) {
      const entries: ReferenceEntry[] = JSON.parse(fs.readFileSync(manifestPath, 'utf-8'))
      const ok = entries.filter(e => e.reference_status === 'ok').length
      const noRef = entries.filter(e => e.reference_status === 'no-reference' || e.reference_status === 'http-404').length
      const failed = entries.filter(e => e.reference_status === 'reference-load-failed').length
      console.log(`\n=== REFERENCE CAPTURE SUMMARY ===`)
      console.log(`Total demos:    ${entries.length}`)
      console.log(`Captured (ok):  ${ok}`)
      console.log(`No reference:   ${noRef}`)
      console.log(`Failed to load: ${failed}`)
    }
  })
})

// Thread-safe manifest update helper (append-safe: re-reads, updates, writes)
function updateManifestEntry(
  manifestPath: string,
  demoId: string,
  updates: Partial<ReferenceEntry>
): void {
  try {
    const entries: ReferenceEntry[] = JSON.parse(fs.readFileSync(manifestPath, 'utf-8'))
    const idx = entries.findIndex(e => e.id === demoId)
    if (idx >= 0) {
      entries[idx] = { ...entries[idx], ...updates }
      fs.writeFileSync(manifestPath, JSON.stringify(entries, null, 2))
    }
  } catch {
    // Non-fatal: capture proceeds even if manifest update fails
  }
}
```
