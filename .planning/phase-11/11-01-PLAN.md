    Port games_fps example demonstrating first-person shooter mechanics.

    Original: https://threejs.org/examples/games_fps.html

    Key features:
    - First-person camera controls
    - Pointer lock for mouse look
    - Movement with WASD
    - Shooting mechanics
    - Enemy/target interaction

    Pattern:
    ```tsx
    export const FPS = () => {
        const cameraRef = $<THREE.PerspectiveCamera>()
        const velocity = $(new THREE.Vector3())
        const direction = $(new THREE.Vector3())
        const moveForward = $(false)
        const moveBackward = $(false)
        const moveLeft = $(false)
        const moveRight = $(false)

        useEffect(() => {
            // Pointer lock
            document.body.requestPointerLock()

            const onKeyDown = (e: KeyboardEvent) => {
                switch (e.code) {
                    case 'KeyW': moveForward(true); break
                    case 'KeyS': moveBackward(true); break
                    case 'KeyA': moveLeft(true); break
                    case 'KeyD': moveRight(true); break
                }
            }

            document.addEventListener('keydown', onKeyDown)
            return () => document.removeEventListener('keydown', onKeyDown)
        }, [])

        useFrame(() => {
            const camera = $$(cameraRef)
            if (!camera) return

            // Calculate movement direction
            const dir = $$(direction)
            dir.z = Number($$(moveForward)) - Number($$(moveBackward))
            dir.x = Number($$(moveRight)) - Number($$(moveLeft))
            dir.normalize()

            // Apply velocity
            camera.position.add(dir.multiplyScalar(0.1))
        })

        return (
            <canvas3D>
                <perspectiveCamera ref={cameraRef} />
                <scene>
                    {/* Level geometry */}
                    {/* Enemies */}
                    {/* Weapons */}
                </scene>
                <pointerLockControls />
            </canvas3D>
        )
    }
    ```

    ```tsx
    import { GLTFExporter } from 'three/examples/jsm/exporters/GLTFExporter'
    import { OBJExporter } from 'three/examples/jsm/exporters/OBJExporter'
    import { STLExporter } from 'three/examples/jsm/exporters/STLExporter'
    import { PLYExporter } from 'three/examples/jsm/exporters/PLYExporter'
    import { DRACOExporter } from 'three/examples/jsm/exporters/DRACOExporter'
    import { USDZExporter } from 'three/examples/jsm/exporters/USDZExporter'

    // GLTF export
    const exportGLTF = (scene: THREE.Scene) => {
        const exporter = new GLTFExporter()
        exporter.parse(
            scene,
            (gltf) => {
                const output = JSON.stringify(gltf, null, 2)
                const blob = new Blob([output], { type: 'application/json' })
                downloadBlob(blob, 'scene.gltf')
            },
            (error) => console.error(error),
            { binary: false }
        )
    }

    // STL export
    const exportSTL = (mesh: THREE.Mesh) => {
        const exporter = new STLExporter()
        const stlString = exporter.parse(mesh, { binary: false })
        const blob = new Blob([stlString], { type: 'text/plain' })
        downloadBlob(blob, 'model.stl')
    }

    // USDZ export (for AR)
    const exportUSDZ = async (scene: THREE.Scene) => {
        const exporter = new USDZExporter()
        const usdz = await exporter.parse(scene)
        const blob = new Blob([usdz], { type: 'application/octet-stream' })
        downloadBlob(blob, 'model.usdz')
    }
    ```

    ```tsx
    <div className="export-buttons">
        <button onClick={() => exportGLTF(scene)}>Export GLTF</button>
        <button onClick={() => exportSTL(mesh)}>Export STL</button>
    </div>
    ```
