  <action>
    Port css2d_label example demonstrating 2D labels in 3D space.

    Original: https://threejs.org/examples/css2d_label.html

    Key features:
    - Labels that follow 3D objects
    - HTML elements positioned in 3D space
    - Label occlusion handling
    - Interactive label content

    Pattern:
    ```tsx
    /** @jsxImportSource @woby/three */
    import { $, $$, useEffect, useFrame } from '@woby/three'
    import { CSS2DRenderer, CSS2DObject } from 'three/examples/jsm/renderers/CSS2DRenderer'

    export const Label = () => {
        const labelRendererRef = $<CSS2DRenderer>()

        useEffect(() => {
            // Create CSS2D renderer
            const labelRenderer = new CSS2DRenderer()
            labelRenderer.setSize(window.innerWidth, window.innerHeight)
            labelRenderer.domElement.style.position = 'absolute'
            labelRenderer.domElement.style.top = '0px'
            document.body.appendChild(labelRenderer.domElement)
            labelRendererRef(labelRenderer)
        }, [])

        // Create label element
        const createLabel = (text: string) => {
            const div = document.createElement('div')
            div.className = 'label'
            div.textContent = text
            div.style.color = 'white'
            div.style.padding = '5px'
            div.style.backgroundColor = 'rgba(0,0,0,0.5)'
            return new CSS2DObject(div)
        }

        useFrame(() => {
            const labelRenderer = $$(labelRendererRef)
            const scene = $$(scenes)?.[0]
            const camera = $$(camera)
            if (labelRenderer && scene && camera) {
                labelRenderer.render(scene, camera)
            }
        })

        return (
            <canvas3D>
                <scene>
                    <mesh>
                        <sphereGeometry args={[1, 32, 32]} />
                        <meshStandardMaterial color="blue" />
                        {/* Label attached to mesh */}
                    </mesh>
                </scene>
            </canvas3D>
        )
    }
    ```

    CRITICAL: CSS2DRenderer renders to a separate DOM element, not WebGL canvas.
    Must handle resize events for both renderers.
  </action>

    ```tsx
    export const Mixed = () => {
        const css3dRendererRef = $<CSS3DRenderer>()
        const webglRendererRef = $<THREE.WebGLRenderer>()

            // CSS3D renderer
            const css3dRenderer = new CSS3DRenderer()
    ```

    ```tsx
    // Create panorama cube faces
    const createPanorama = () => {
        const sides = ['right', 'left', 'top', 'bottom', 'front', 'back']
        const group = new THREE.Group()

        sides.forEach((side, i) => {
            const element = document.createElement('div')
            element.style.width = '1024px'
            element.style.height = '1024px'
            element.style.background = `url(${side}.jpg)`
            element.style.backgroundSize = 'cover'

            const object = new CSS3DObject(element)
            object.position.set(
                i === 0 ? 512 : i === 1 ? -512 : 0,
                i === 2 ? 512 : i === 3 ? -512 : 0,
                i === 4 ? 512 : i === 5 ? -512 : 0
            )
            object.rotation.y = i === 0 ? -Math.PI/2 : i === 1 ? Math.PI/2 : 0
            group.add(object)
        })

        return group
    }
    ```
